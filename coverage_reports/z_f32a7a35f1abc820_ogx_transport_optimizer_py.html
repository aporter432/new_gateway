61class TransportOptimizer: 

62 """OGx transport optimization service. 

63 

64 This class implements intelligent transport selection based on: 

65 - Current network conditions 

66 - Message characteristics 

67 - Terminal capabilities 

68 - Cost considerations

71 def __init__(self, network_monitor: NetworkMonitor): 

72 """Initialize transport optimizer. 

73 

74 Args: 

75 network_monitor: Network condition monitor 

76 """ 

77 self.network_monitor = network_monitor 

78 self.logger = get_protocol_logger(LoggingConfig()) 

79 self.settings = get_settings() 

80 

81 # Transport-specific thresholds 

82 self._error_rate_threshold = 20.0 # Consider transport degraded above 20% errors 

83 self._latency_threshold_ms = 5000 # Consider transport degraded above 5s latency 

84 self._availability_threshold = 80.0 # Consider transport degraded below 80% availability

86 async def determine_transport( 

87 self, 

88 message: Dict, 

89 terminal_id: str, 

90 network_type: NetworkType, # Kept for API compatibility 

91 preferred_transport: Optional[TransportType] = None, 

92 ) -> TransportType: 

93 """Determine optimal transport for message. 

94 

95 Implements transport selection following OGx-1.txt Section 5.4: 

96 1. Check message constraints (size, priority) 

97 2. Check terminal capabilities 

98 3. Evaluate network conditions 

99 4. Apply cost optimization 

100 

101 Args: 

102 message: Message to be sent 

103 terminal_id: Destination terminal ID 

104 network_type: Network type (OGx or IDP) 

105 preferred_transport: Optional preferred transport type 

106 

107 Returns: 

108 Selected transport type 

109 

110 Raises: 

111 ValidationError: If no suitable transport available 

112 """ 

113 # Get message size 

114 message_size = len(str(message).encode()) # Simple size estimation 

115 

116 # Validate message size against network limits 

117 if not self._is_size_valid_for_network(message_size): 

118 raise ValidationError( 

119 f"Message size {message_size} bytes exceeds network limit", 

120 error_code=ValidationError.INVALID_MESSAGE_FORMAT, 

121 ) 

122 

123 # Get available transports based on network type 

124 available_transports = self._get_available_transports() 

125 

126 # If preferred transport is specified and available, use it 

127 if preferred_transport in available_transports: 

128 return preferred_transport 

129 

130 # Get network metrics for available transports 

131 transport_metrics = [] 

132 for transport in available_transports: 

133 metrics = await self.network_monitor.get_network_metrics(transport) 

134 if await self._is_transport_healthy(metrics): 

135 transport_metrics.append((transport, metrics)) 

136 

137 if not transport_metrics: 

138 raise ValidationError( 

139 "No healthy transports available", error_code=ValidationError.INVALID_MESSAGE_FORMAT 

140 ) 

141 

142 # Select best transport based on conditions and cost 

143 selected_transport = await self._select_best_transport(transport_metrics) 

144 

145 self.logger.info( 

146 "Transport selected", 

147 extra={ 

148 "terminal_id": terminal_id, 

149 "message_size": message_size, 

150 "network_type": network_type.name, 

151 "selected_transport": selected_transport.name, 

152 "available_transports": [t.name for t, _ in transport_metrics], 

153 }, 

154 ) 

155 

156 return selected_transport

158 def _is_size_valid_for_network(self, message_size: int) -> bool: 

159 """Check if message size is valid for network.""" 

160 return message_size <= MAX_OGX_PAYLOAD_BYTES 

161 

162 def _get_available_transports(self) -> List[TransportType]: 

163 """Get list of available transport types.""" 

164 return [TransportType.SATELLITE, TransportType.CELLULAR] 

165 

166 async def _is_transport_healthy(self, metrics: NetworkMetrics) -> bool: 

167 """Check if transport is healthy based on metrics. 

168 

169 Args: 

170 metrics: Network metrics for transport 

171 

172 Returns: 

173 True if transport is healthy 

174 """ 

175 # Check if we have recent metrics 

176 if metrics.timestamp < datetime.utcnow() - timedelta(minutes=5): 

177 return False 

178 

179 # Check error rate 

180 if metrics.error_rate and metrics.error_rate > self._error_rate_threshold: 

181 return False 

182 

183 # Check latency 

184 if metrics.latency_ms and metrics.latency_ms > self._latency_threshold_ms: 

185 return False 

186 

187 # Check availability 

188 if metrics.availability and metrics.availability < self._availability_threshold: 

189 return False 

190 

191 return True

193 async def _select_best_transport( 

194 self, transport_metrics: List[Tuple[TransportType, NetworkMetrics]] 

195 ) -> TransportType: 

196 """Select best transport based on conditions and cost. 

197 

198 Args: 

199 transport_metrics: List of (transport, metrics) tuples 

200 

201 Returns: 

202 Selected transport type 

203 """ 

204 # Simple scoring system (higher is better) 

205 scores: Dict[TransportType, float] = {} 

206 

207 for transport, metrics in transport_metrics: 

208 base_score = 100.0 

209 

210 # Penalize for high error rate 

211 if metrics.error_rate: 

212 base_score -= metrics.error_rate 

213 

214 # Penalize for high latency 

215 if metrics.latency_ms: 

216 base_score -= (metrics.latency_ms / self._latency_threshold_ms) * 20 

217 

218 # Bonus for high availability 

219 if metrics.availability: 

220 base_score += (metrics.availability / 100.0) * 20 

221 

222 # Prefer cellular for cost optimization if conditions similar 

223 if transport == TransportType.CELLULAR: 

224 base_score += 10 

225 

226 scores[transport] = base_score 

227 

228 # Select transport with highest score 

229 return max(scores.items(), key=lambda x: x[1])[0]
